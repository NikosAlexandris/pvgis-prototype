import requests
latitude = 45
longitude = 8.6

response = requests.get("https://pvgis6.jrc.it/performance/broadband", params={
    "latitude": latitude,
    "longitude": longitude,
    "start_time": "2005-01-01 00:00:00",
    "end_time": "2020-12-31 23:59:59"
})

print(response.json())
                
fetch("https://pvgis6.jrc.it/performance/broadband?\
lat=45&lon=8.6&start_time=2005-01-01%2000%3A00%3A00&end_time=2020-12-31%2023%3A59%3A59")
.then(res => res.json())
.then(data => console.log(data));
