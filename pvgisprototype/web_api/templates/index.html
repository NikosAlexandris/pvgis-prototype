import requests

latitude = 45.812
longitude = 8.628

response = requests.get("https://pvgis6.jrc.it/performance/broadband", params={
    "latitude": latitude,
    "longitude": longitude,
    "start_time": "2005-01-01 00:00:00",
    "end_time": "2020-12-31 23:59:59"
})

print(response.json())
const url = new URL('https://pvgis6.jrc.it/performance/broadband');
url.searchParams.set('latitude', '45.812');
url.searchParams.set('longitude', '8.628');
url.searchParams.set('start_time', '2005-01-01 00:00:00');
url.searchParams.set('end_time', '2020-12-31 23:59:59');

fetch(url)
  .then(res => res.json())
  .then(data => console.log(data));
